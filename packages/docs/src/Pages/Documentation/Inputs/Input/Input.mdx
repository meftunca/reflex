import { TextField, Button, Box } from "@re-flex/ui";

## Basic usage

### Example

```js live=true column=true
<>
  <Box bgColor="#0001" radius={[16, 16, 0, 0]} width="auto" pt={8}>
    <TextField fullWidth startAdornment={<Button>H</Button>} />
  </Box>
  <TextField tag="textarea" startAdornment={<Button>H</Button>} />
</>
```

### Filled

```js live=true column=true
<>
  <Box bgColor="#0001" radius={[16, 16, 0, 0]} width="auto" pt={8}>
    <TextField fullWidth startAdornment={<Button>H</Button>} />
  </Box>
  <TextField tag="textarea" endAdornment={<Button>H</Button>} />
  <TextField tag="div" endAdornment={<Button>H</Button>} />
</>
```

### Outlined

```js live=true  column=true
<>
  <TextField variant="outlined" startAdornment={<Button>H</Button>} />
  <TextField
    variant="outlined"
    tag="textarea"
    endAdornment={<Button>H</Button>}
  />
  <TextField variant="outlined" tag="div" endAdornment={<Button>H</Button>} />
</>
```
