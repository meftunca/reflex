import { Box, Button, ButtonBase } from "@re-flex/ui";

# With Icon

### Example Usage

```js live=true
<>
  <ButtonBase p={[12, 24]} as="a">
    Based Button
  </ButtonBase>
  <Button
    variant="outlined"
    affixIcon={<FiCheckCircle />}
    color="success"
    size={14}
  >
    Icon Check Circle (affix)
  </Button>

  <Button
    variant="outlined"
    suffixIcon={<FiAlertOctagon />}
    color="error"
    size={14}
  >
    Icon Alert (suffix)
  </Button>
</>
```

# Sizes

### Example Usage

```js live=true
<>
  <Button variant="contained" color="primary" size={10}>
    Primary 10px
  </Button>

  <Button variant="contained" color="primary" size={12}>
    Primary 12px
  </Button>

  <Button variant="contained" color="primary" size={14}>
    Primary 14px
  </Button>
</>
```
